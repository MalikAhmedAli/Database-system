create database organizationn;
use organizationn;


create TABLE EMPLOYEE(
     EMP_NUM   NUMERIC(4) PRIMARY KEY,
     BIRTH_DATA DATE NOT NULL,
     FIRST_NAME VARCHAR(14),
     LAST_NAME VARCHAR(16),
     SEX ENUM('M','F'),
     HIRE_DATE DATE NOT NULL);

create table DEPARTMENTS(
     DEPT_NO  CHAR(4) PRIMARY KEY,
     DEPT_NAME  VARCHAR(40));


create table DEPT_MANAGER(
     DEPT_NO  CHAR(4),
     EMP_NUM  NUMERIC(4),
     INDEX    (EMP_NUM),
     INDEX    (DEPT_NO),
     FROM_DATE  DATE NOT NULL,
     TO_DATE   DATE NOT NULL,
     CONSTRAINT  FK_DEPT_EMP FOREIGN KEY(EMP_NUM)REFERENCES EMPLOYEE(EMP_NUM),
     CONSTRAINT  FK_DEPT_EMP_DEPTS FOREIGN KEY(DEPT_NO)REFERENCES DEPARTMENTS(DEPT_NO));

create table  DEPT_EMP(
     EMP_NUM  NUMERIC(4),
     INDEX  (EMP_NUM),
     DEPT_NO  CHAR(4),
     INDEX  (DEPT_NO),
     FROM_DATE DATE NOT NULL,
     TO_DATE DATE NOT NULL,
     CONSTRAINT FK_DEPT_EMP_EMPL FOREIGN KEY (EMP_NUM)REFERENCES EMPLOYEE(EMP_NUM),
     CONSTRAINT FK_DEPT_EMP_DEPT FOREIGN KEY (DEPT_NO)REFERENCES DEPARTMENTS(DEPT_NO));


create table TITLES(
    -> EMP_NUM  NUMERIC(4),
    -> TITLE    VARCHAR(50) PRIMARY KEY,
    -> FROM_DATE DATE NOT NULL,
    -> TO_DATE   DATE NOT NULL,
    -> INDEX    (EMP_NUM),
    -> CONSTRAINT  FK_TITLES_EMP FOREIGN KEY(EMP_NUM)REFERENCES EMPLOYEE(EMP_NUM));

create table SALARIES(
    -> EMP_NUM  NUMERIC(4),
    -> INDEX  (EMP_NUM),
    -> SALARY NUMERIC(4,2)DEFAULT 00.00 NOT NULL,
    -> FROM_DATE DATE NOT NULL,
    -> TO_DATE DATE NOT NULL,
    -> CONSTRAINT PK_SALARIES PRIMARY KEY(FROM_DATE,TO_DATE),
    -> CONSTRAINT  FK_SALARIES_EMP FOREIGN KEY(EMP_NUM)REFERENCES EMPLOYEE(EMP_NUM));
